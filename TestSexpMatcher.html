def test_cls_m
  re = Regexp.union [/\w/, /\d/]

  assert_equal M::Pattern.new(/a/),       s{ m(/a/)       }
  assert_equal M::Pattern.new(/\Aa\Z/),   s{ m(:a)        }
  assert_equal M::Pattern.new(/test_\w/), s{ m(/test_\w/) }
  assert_equal M::Pattern.new(re),        s{ m(/\w/,/\d/) }
end
