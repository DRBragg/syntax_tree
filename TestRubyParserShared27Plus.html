def test_defn_no_kwargs
  # def x(**nil)
  # end
  #
  # def y(**kw)
  # end
  #
  # def z()
  # end
  #
  # x arg: 42 # $!: no keywords accepted (ArgumentError)
  # y arg: 42 # fine
  # z arg: 42 # $!: wrong number of arguments (given 1, expected 0) (ArgumentError)

  rb = "def x(**nil); end"
  pt = s(:defn, :x, s(:args, :"**nil"),
         s(:nil))

  assert_parse rb, pt
end

def test_mlhs_rescue
  # same:
  # a = (24 rescue 42)
  # a = 24 rescue 42

  # same:
  # a, b = (f rescue 42)
  # a, b = f rescue 42

  rb = "a, b = f rescue 42"
  pt = s(:masgn,
         s(:array, s(:lasgn, :a), s(:lasgn, :b)),
         s(:to_ary,
           s(:rescue,
             s(:call, nil, :f),
             s(:resbody, s(:array),
               s(:lit, 42)))))

  assert_parse rb, pt
end
